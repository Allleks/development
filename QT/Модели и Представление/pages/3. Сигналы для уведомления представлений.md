### При изменении данных:
```cpp
void dataChanged(const QModelIndex &topLeft, const QModelIndex &bottomRight,
                const QVector<int> &roles = QVector<int>());

// Для одного элемента:
void dataChanged(const QModelIndex &index, const QVector<int> &roles);
```
### При структурных изменениях:
```cpp
// Перед добавлением:
void beginInsertRows(const QModelIndex &parent, int first, int last);
void beginInsertColumns(const QModelIndex &parent, int first, int last);

// После добавления:
void endInsertRows();
void endInsertColumns();

// Перед удалением:
void beginRemoveRows(const QModelIndex &parent, int first, int last);
void beginRemoveColumns(const QModelIndex &parent, int first, int last);

// После удаления:
void endRemoveRows();
void endRemoveColumns();

// Сброс модели (полная перезагрузка):
void beginResetModel();
void endResetModel();

// Перемещение строк:
void beginMoveRows(const QModelIndex &sourceParent, int sourceFirst, int sourceLast,
                   const QModelIndex &destinationParent, int destinationChild);
void endMoveRows();
```
### При изменении заголовков:
```cpp
void headerDataChanged(Qt::Orientation orientation, int first, int last);
```
