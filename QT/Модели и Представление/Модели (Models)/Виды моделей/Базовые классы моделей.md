### **1. QAbstractItemModel** (самый гибкий)
```cpp
// Для любых структур: деревья, таблицы, списки
class CustomTreeModel : public QAbstractItemModel {
    // Обязательные методы:
    QModelIndex index(int row, int column, 
                     const QModelIndex& parent) const override;
    QModelIndex parent(const QModelIndex& index) const override;
    int rowCount(const QModelIndex& parent) const override;
    int columnCount(const QModelIndex& parent) const override;
    QVariant data(const QModelIndex& index, int role) const override;
    
    // Использовать для: древовидных структур, сложных моделей
};
```
### **2. QAbstractTableModel** (для таблиц)
```cpp
// Только таблицы, НЕТ иерархии
class CustomTableModel : public QAbstractTableModel {
    // Упрощенный интерфейс:
    int rowCount(const QModelIndex& parent = QModelIndex()) const override;
    int columnCount(const QModelIndex& parent = QModelIndex()) const override;
    QVariant data(const QModelIndex& index, int role) const override;
    
    // parent всегда невалидный (нет иерархии)
    QModelIndex parent(const QModelIndex&) const override { 
        return QModelIndex(); 
    }
    
    // Использовать для: табличных данных, CSV, Excel-подобных данных
};
```
### **3. QAbstractListModel** (для списков)
```cpp
// Только один столбец, НЕТ иерархии
class CustomListModel : public QAbstractListModel {
    // Самый простой интерфейс:
    int rowCount(const QModelIndex& parent = QModelIndex()) const override;
    QVariant data(const QModelIndex& index, int role) const override;
    
    // columnCount всегда 1
    int columnCount(const QModelIndex&) const override { 
        return 1; 
    }
    
    // Использовать для: списков, комбобоксов, QListView
};
```
### **4. QStandardItemModel** (готовый к использованию)
```cpp
// Уже реализованы все методы, работайте с QStandardItem
QStandardItemModel* model = new QStandardItemModel();

// Добавляем данные:
QStandardItem* item = new QStandardItem("Text");
item->setIcon(QIcon(":/icon.png"));
item->setData(42, Qt::UserRole);
model->appendRow(item);

// Использовать для: простых случаев, прототипирования
// НЕ использовать для: больших данных, кастомного поведения
```
### **5. QSqlQueryModel / QSqlTableModel** (для БД)
```cpp
// Готовые модели для работы с SQL
QSqlTableModel* model = new QSqlTableModel();
model->setTable("operations");
model->select(); // Загружает данные из БД

// QSqlTableModel - можно редактировать

// Использовать для: быстрого подключения к БД
```
