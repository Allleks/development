## Progress Bar Delegate
```cpp
class ProgressBarDelegate : public QStyledItemDelegate {
public:
    void paint(QPainter* painter, const QStyleOptionViewItem& option,
               const QModelIndex& index) const override {
        
        // Получаем значение из модели
        int progress = index.data(Qt::DisplayRole).toInt();
        
        // Рисуем прогресс-бар
        QStyleOptionProgressBar progressBarOption;
        progressBarOption.rect = option.rect;
        progressBarOption.minimum = 0;
        progressBarOption.maximum = 100;
        progressBarOption.progress = progress;
        progressBarOption.text = QString::number(progress) + "%";
        progressBarOption.textVisible = true;
        
        // Используем стиль системы
        QApplication::style()->drawControl(QStyle::CE_ProgressBar,
                                          &progressBarOption, painter);
    }
    
    QSize sizeHint(const QStyleOptionViewItem& option,
                   const QModelIndex& index) const override {
        return QSize(100, 20); // Фиксированный размер
    }
};

// Использование:
tableView->setItemDelegateForColumn(2, new ProgressBarDelegate(tableView));
```

## ComboBox Delegate
```cpp
class ComboBoxDelegate : public QStyledItemDelegate {
public:
    ComboBoxDelegate(const QStringList& items, QObject* parent = nullptr)
        : QStyledItemDelegate(parent), m_items(items) {}
    
    QWidget* createEditor(QWidget* parent,
                         const QStyleOptionViewItem& option,
                         const QModelIndex& index) const override {
        QComboBox* editor = new QComboBox(parent);
        editor->addItems(m_items);
        return editor;
    }
    
    void setEditorData(QWidget* editor,
                      const QModelIndex& index) const override {
        QComboBox* comboBox = static_cast<QComboBox*>(editor);
        QString currentText = index.data(Qt::DisplayRole).toString();
        comboBox->setCurrentText(currentText);
    }
    
    void setModelData(QWidget* editor,
                     QAbstractItemModel* model,
                     const QModelIndex& index) const override {
        QComboBox* comboBox = static_cast<QComboBox*>(editor);
        model->setData(index, comboBox->currentText(), Qt::EditRole);
    }
    
private:
    QStringList m_items;
};

// Использование:
QStringList types = {"Admin", "User", "Guest"};
tableView->setItemDelegateForColumn(3, new ComboBoxDelegate(types, tableView));
```
