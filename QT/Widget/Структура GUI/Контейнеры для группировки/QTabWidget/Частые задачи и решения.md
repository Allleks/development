### Задача 1: Динамическое создание вкладок
```cpp
void addDynamicTab(const QString& title, const QString& content)
{
    QTextEdit* editor = new QTextEdit();
    editor->setPlainText(content);
    
    int index = tabWidget->addTab(editor, title);
    tabWidget->setCurrentIndex(index);
}
```
### Задача 2: Сохранение данных перед закрытием
```cpp
void closeEvent(QCloseEvent* event)
{
    // Проверяем все вкладки на несохраненные изменения
    for (int i = 0; i < tabWidget->count(); i++) {
        TextEditor* editor = qobject_cast<TextEditor*>(tabWidget->widget(i));
        if (editor && editor->hasUnsavedChanges()) {
            if (!askSaveChanges()) {
                event->ignore(); // Не закрываем
                return;
            }
        }
    }
    
    event->accept(); // Закрываем
}
```
### Задача 3: Свои заголовки вкладок
```cpp
// Заменяем стандартный заголовок на свою кнопку
QWidget* createCustomTabHeader(const QString& title)
{
    QWidget* widget = new QWidget();
    QHBoxLayout* layout = new QHBoxLayout(widget);
    layout->setContentsMargins(0, 0, 0, 0);
    
    QLabel* label = new QLabel(title);
    QPushButton* closeButton = new QPushButton("×");
    closeButton->setFixedSize(16, 16);
    
    layout->addWidget(label);
    layout->addWidget(closeButton);
    
    // Подключаем кнопку закрытия
    connect(closeButton, &QPushButton::clicked, [this, widget]() {
        int index = tabWidget->indexOf(widget);
        if (index >= 0) {
            tabWidget->removeTab(index);
        }
    });
    
    return widget;
}

// Использование:
tabWidget->setTabBar(new CustomTabBar());
```
