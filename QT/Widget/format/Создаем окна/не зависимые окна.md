по текущему сигналу можно создавать много окон
```cpp
QtWidgets::QtWidgets(QWidget *parent)
    : QMainWindow(parent)
{
    ui.setupUi(this);

    QVBoxLayout layout(this);
    QPushButton* btn = new QPushButton("Open new window", this);
    //btn->resize(200, 100);
    btn->setGeometry(150,30,150,80);

    layout.addWidget(btn);

    QObject::connect(btn, &QPushButton::clicked, [&]() {
      // auto window = std::make_unique<QWidget>(); с умным указателем не работает
        QWidget* window = new QWidget();

		//при закрытии окна вызовет delete и корректно осободит память
        window->setAttribute(Qt::WA_DeleteOnClose);
        
        window->setWindowTitle("Operation");
        window->resize(800, 600);
        window->show();
        //потому что уничтожаеться здесь окно, тоесть быстро показалось и уничтожилось
    });
}
```
