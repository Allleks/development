```cpp
void EditExpressionWindow::setupExpressionTables()
{
	// Настройка таблицы 1
	ui.expression_table_1->setColumnCount(7);
	ui.expression_table_1->setHorizontalHeaderLabels({
		"Код", "Название", "Описание", "Ед.изм",
		"Мин. знач", "Макс.знач", "Тест. знач"
													 });

	// Настройка таблицы 2
	ui.expression_table_2->setColumnCount(7);
	ui.expression_table_2->setHorizontalHeaderLabels({
		"Код", "Название", "Описание", "Ед.изм",
		"Мин. знач", "Макс.знач", "Тест. знач"
													 });

	// Общие настройки для обеих таблиц
	for (auto* table : {ui.expression_table_1, ui.expression_table_2}) {
		table->horizontalHeader()->setSectionResizeMode(QHeaderView::Stretch);
	}
}

void EditExpressionWindow::fillExpressionTables()
{
	auto manager = std::make_unique<mcc::TestRepositoriesManager>();
	const auto& property = manager->equipmentProperties;

	int rowCount = static_cast<int>(property.size());
	ui.expression_table_1->setRowCount(rowCount);
	ui.expression_table_2->setRowCount(rowCount);

	int row = 0;
	for (const auto& [key, value] : property)
	{
		if (!value) continue;

		// Для каждой таблицы
		for (auto* table : {ui.expression_table_1, ui.expression_table_2}) {
			// Колонки 0-5: НЕредактируемые
			table->setItem(row, 0, createReadOnlyItem(QString::fromStdString(key)));
			table->setItem(row, 1, createReadOnlyItem(QString::fromLocal8Bit(value->GetName())));
			table->setItem(row, 2, createReadOnlyItem(QString::fromLocal8Bit(value->GetDescription())));
			table->setItem(row, 3, createReadOnlyItem("кгс/см²"));
			table->setItem(row, 4, createReadOnlyItem(""));
			table->setItem(row, 5, createReadOnlyItem(""));

			// Колонка 6: редактируемая
			QTableWidgetItem* editableItem = new QTableWidgetItem("");
			editableItem->setFlags(editableItem->flags() | Qt::ItemIsEditable);
			table->setItem(row, 6, editableItem);
		}

		row++;
	}
}

QTableWidgetItem* EditExpressionWindow::createReadOnlyItem(const QString& text)
{
	QTableWidgetItem* item = new QTableWidgetItem(text);

	item->setFlags(item->flags() & ~Qt::ItemIsEditable);

	return item;
}
```
**Ключевые изменения:**
1. Все колонки 0-5: `& ~Qt::ItemIsEditable` (ОТКЛЮЧАЕМ редактирование)
2. Только колонка 6: `| Qt::ItemIsEditable` (ВКЛЮЧАЕМ редактирование)
