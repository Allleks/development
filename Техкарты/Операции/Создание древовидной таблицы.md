void OperationsTreeViewModel::createOperationTreeViewItems(const std::shared_ptr<mcc::Operation>& operation)
{
    try
    {
        auto groupId = operation->GetOperationsGroup()->GetId();

        auto groupIterator = std::ranges::find_if(
            rootItem->getChildren().cbegin(),
            rootItem->getChildren().cend(),
            [groupId](const std::unique_ptr<AbstractTreeViewItem>& group) { return group->getId() == groupId; });

        GroupOperationsTreeViewItem* groupItem = nullptr;

        if (groupIterator == rootItem->getChildren().end())
        {
            auto newGroupoItem = std::make_unique<GroupOperationsTreeViewItem>(
                operation->GetOperationsGroup(),
                dynamic_cast<HeaderTreeViewItem*>(rootItem.get()));

            groupItem = newGroupoItem.get();

            rootItem->appendChild(std::move(newGroupoItem));
        }
        else
        {
            groupItem = dynamic_cast<GroupOperationsTreeViewItem*>(groupIterator->get());
        }

        if (groupItem == nullptr)
        {
            qCritical() << "groupItem IS NULL";

            return;
        }

        auto newOperationTreeViewItem = std::make_unique<OperationOperationsTreeViewItem>(operation, groupItem);

        for (const auto& [sparePartId, sparePart] : operation->GetSpareParts())
        {
            if (const auto hardcodedSparePart = std::dynamic_pointer_cast<mcc::OperationItemHardcodedSparePart>(sparePart); hardcodedSparePart != nullptr)
            {
                newOperationTreeViewItem->appendChild(
                    std::make_unique<HardcodedSparePartOperationsTreeViewItem>(hardcodedSparePart.get(), newOperationTreeViewItem.get()));
            }
            else
            {
                newOperationTreeViewItem->appendChild(
                    std::make_unique<SparePartOperationsTreeViewItem>(sparePart.get(), newOperationTreeViewItem.get()));
            }
        }

        for (const auto& [sparePartId, sparePart] : operation->GetMachineries())
        {
            newOperationTreeViewItem->appendChild(std::make_unique<MachineryOperationsTreeViewItem>(sparePart.get(), newOperationTreeViewItem.get()));
        }

        groupItem->appendChild(std::move(newOperationTreeViewItem));
    }
    catch (const std::exception& ex)
    {
        qCritical() << ex.what();
    }
}
