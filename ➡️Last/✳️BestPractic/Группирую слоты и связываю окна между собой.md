void OperationsGroupWindow::setupExpressionButton(QPushButton* button, QLabel* label, QTextEdit* textEdit)
{
	connect(button, &QPushButton::clicked, this, [this, label, textEdit]() {
		auto editExpressionWindow = new EditExpressionWindow(
			label->text()
		);

		connect(editExpressionWindow, &EditExpressionWindow::expressionResultReady,
				this, [this, textEdit](const QString& result){
					textEdit->setText(result);
		});

		editExpressionWindow->show();
	});
}
